}; Roo.extend(Roo.bootstrap.NavProgressBar, Roo.bootstrap.Component, { bullets : [], barItems : [], getAutoCreate : function() { var cfg = Roo.apply({}, Roo.bootstrap.NavProgressBar.superclass.getAutoCreate.call(this)); } index = i; return false }); return index; }, setActiveNext : function() { var i = this.indexOfItem(this.getActive()); if (i > this.barItems.length) { return; } this.setActiveItem(this.barItems[i+1]);
