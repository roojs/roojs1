}; Roo.extend(Roo.bootstrap.FieldLabel, Roo.bootstrap.Component, { tag: 'label', cls: '', html: '', target: '', allowBlank : true, invalidClass : 'text-danger fa fa-lg fa-exclamation-triangle', validClass : 'text-success fa fa-lg fa-check', iconTooltip : 'This field is required', getAutoCreate : function(){ var cfg = { tag : this.tag, cls : 'roo-bootstrap-field-label ' + this.cls, for : this.target, cn : [ { tag : 'i', cls : '', tooltip : this.iconTooltip }, { tag : 'span', html : this.html } ] };
